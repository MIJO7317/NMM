u_0 = lambda x: np.sin(np.pi * x)

#   3 
# alpha_1 * u_x(0, t) + beta_1 * u(0, t) = gamma_1(t)
# alpha_2 * u_x(l, t) + beta_2 * u(l, t) = gamma_2(t)
alpha_1 = 0
alpha_2 = 0
beta_1 = 1
beta_2 = 1
gamma_1 = lambda t: 0
gamma_2 = lambda t: 0

f = lambda x, t: 0
k = lambda u, x, t: 1

#  
N = 101
M = 101
xs = np.linspace(0, l, N)
ts = np.linspace(0, T, M)
h = xs[1] - xs[0]
tau = ts[1] - ts[0]

#   
u = np.zeros((M, N))
u[0] = u_0(xs)

#   
for m in range(M - 1):
    #   A   b
    A = np.zeros((N, N))
    b = np.zeros(N)
    A[0][0] = beta_1 - 3/2 * alpha_1 / h
    A[0][1] = 2 * alpha_1 / h
    A[0][2] = -alpha_1 / (2 * h)
    b[0] = gamma_1(ts[m])
    A[N - 1][N - 3] = alpha_2 / (2 * h)
    A[N - 1][N - 2] = -2 * alpha_2 / h
    A[N - 1][N - 1] = beta_2 + 3/2 * alpha_2 / h
    b[N - 1] = gamma_2(ts[m])
    for n in range(1, N - 1):
        A[n][n - 1] = -1 / h ** 2 * (k(u[m][n], xs[n], ts[m]+tau/2) + k(u[m][n-1], xs[n-1], ts[m]+tau/2)) / 2
        A[n][n] = 2 / h ** 2 * k(u[m][n], xs[n], ts[m]+tau/2) + 1 / tau
        A[n][n + 1] = -1 / h ** 2 * (k(u[m][n], xs[n], ts[m]+tau/2) + k(u[m][n+1], xs[n+1], ts[m]+tau/2)) / 2
        b[n] = u[m][n] / tau + f(xs[n], ts[m])
    #  
    u[m + 1] = np.linalg.solve(A, b)
